4.实现一个全屏自适应的遮罩层
    {
        position:absolute;
        width:100%;
        height:100%;
        left:0;
        top:0;
    }
    {
        position:absolute;
        left:0;
        top:0;
        right:0;
        bottom:0;
    }
5.实现一个距离右侧100px的全屏自适应容器层
    {
        position:absolute;
        left:0;
        top:0;
        bottom:0;
        right:100px;
    }
    {
        position:absolute;
        left:0;
        top:0;
        bottom:0;
        width:calc(100%-100px)
    }
